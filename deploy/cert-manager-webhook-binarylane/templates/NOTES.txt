Thank you for installing cert-manager-webhook-binarylane!

The webhook is now running. Create a ClusterIssuer to use it:

cat <<EOF | kubectl apply -f -
apiVersion: cert-manager.io/v1
kind: ClusterIssuer
metadata:
  name: letsencrypt-binarylane
spec:
  acme:
    email: your-email@example.com
    server: https://acme-v02.api.letsencrypt.org/directory
    privateKeySecretRef:
      name: letsencrypt-binarylane-account-key
    solvers:
    - dns01:
        webhook:
          groupName: {{ .Values.groupName }}
          solverName: binarylane
          config:
            apiKeySecretRef:
              name: binarylane-api-credentials
              key: api-key
EOF
